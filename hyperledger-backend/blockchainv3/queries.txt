query selectImageById {
  description: "Select Image by ID"
  statement:
      SELECT org.blockchain.health.Image
          WHERE (id_img==_$id_img)
}

query selectAssetByDoctor {
	description: "Select an asset based on the owner"
	statement:
		SELECT org.blockchain.health.Image
			WHERE (doc == _$doc)
}
