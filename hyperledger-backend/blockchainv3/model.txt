namespace org.blockchain.health

/** Enums **/

enum Status {
 o diagnosed
 o undiagnosed
}

/** Participants **/

participant Doctor identified by id_doc {
  o String id_doc
  o String name optional
  o String address optional
}

participant MLA identified by id_mla {
  o String id_mla
  o String name optional
  o String address optional
}

/** Assets **/

asset Image identified by id_img {
  o String id_img
  --> Doctor doc
  --> MLA mla optional
  o Status status default="undiagnosed"
  o Boolean cancer optional
  o Integer width optional
  o Integer height optional
  o String img_base64
  o String img_csv optional
}

/** Transactions & Events **/

/** Doctor **/
  
transaction create_img {
  o String id_doc
  o String id_img
  o Integer width optional
  o Integer height optional
  o String img_base64
  o String img_csv optional
}
  
event create_img_event{
  --> Doctor doc
}

/** Machine Learning Architect **/
  
transaction diagnose_img {  
  o String id_mla
  o String id_img
  o Boolean cancer
}
  
event diagnose_img_event{
  --> Image img_base64
  --> MLA mla
  o Status status
  o Boolean cancer
}
